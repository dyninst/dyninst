\begin{apient}
  BPatch_constExpr(signed int value)
  BPatch_constExpr(unsigned int value)
  BPatch_constExpr(signed long value)
  BPatch_constExpr(unsigned long value)
  BPatch_constExpr(const char *value)
  BPatch_constExpr(const void *value)
  BPatch_constExpr(long long value)
\end{apient}

\apidesc{

  Define a constant snippet of the appropriate type. The \code{char\*} form
  of the constructor creates a constant string; the null-terminated
  string beginning at the location pointed to by the parameter is copied
  into the application's address space, and the \code{BPatch\_constExpr}
  that is created refers to the location to which the string was copied.
}
