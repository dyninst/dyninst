\begin{apient}
  BPatch_whileExpr(const BPatch_snippet &condition, const BPatch_snippet &body)
\end{apient}

\apidesc{
  This constructor creates a while statement. The first argument,
  condition, should be a Boolean expression that will be
  evaluated to decide whether body should be executed. The second
  argument, body, is the snippet to execute if the
  condition evaluates to true.
}
