\begin{apient}
  BPatch_actualAddressExpr()
\end{apient}

\apidesc{
  This snippet results in an expression that evaluates to the actual
  address of the instrumentation. To access the
  original address where instrumentation was inserted, use
  \code{BPatch\_originalAddressExpr}. Note that this actual address is
  highly dependent on a number of internal variables and has no
  relation to the original address.
}
