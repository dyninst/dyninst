\begin{apient}
  BPatch_breakPointExpr()
\end{apient}

\apidesc{
  Define a snippet that stops a process when executed by it. The stop
  can be detected using the isStopped member
  function of \code{BPatch\_process,}  and the program's execution
  can be resumed by calling the
  continueExecution member function of \code{BPatch\_process.}
}
